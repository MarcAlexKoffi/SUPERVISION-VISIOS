  formatEval(val: string): string {
      if (!val) return '-';
      const map: any = {
          'conforme': 'Conforme', 'a_ameliorer': 'À améliorer', 'non_conforme': 'Non Conforme',
          'excellent': 'Excellent', 'satisfaisant': 'Satisfaisant', 'insuffisant': 'Insuffisant'
      };
      return map[val] || val;
  }
